#include <iostream>
#include <cmath>

using namespace std;

// Умова закінчення
bool converge(double xk[10], double xkp[10], int n, double eps)
{
	double norm = 0;
	for (int i = 0; i < n; i++)
		norm += (xk[i] - xkp[i]) * (xk[i] - xkp[i]);
	return (sqrt(norm) < eps);
}

double okr(double x, double eps)
{
	int i = 0;
	double neweps = eps;
	while (neweps < 1)
	{
		i++;
		neweps *= 10;
	}
	int okr = pow(double(10), i);
	x = int(x * okr + 0.5) / double(okr);

	return x;
}

bool diagonal(double a[3][3])
{
	int i, j, k = 1;
	double sum;
	for (i = 0; i < 3; i++) {
		sum = 0;
		for (j = 0; j < 3; j++) sum += abs(a[i][j]);
		sum -= abs(a[i][i]);
		if (sum > a[i][i])
		{
			k = 0;
			cout << a[i][i] << " < " << sum << endl;
		}
		else
		{
			cout << a[i][i] << " > " << sum << endl;
		}


	}

	return (k == 1);

}

int main()
{

	double eps = 0.1, x[3], p[3];
	double a[3][3] = {
			{10, -1, -2},
			{-1, -3, -15},
			{-2, -15, 9},
	};
	double b[3] = {7, -19, -8};
	int i, j, m = 0;
	int method;
	
	cout << endl << endl;
	cout << "Matrix A: " << endl << endl;
	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 3; j++)
			cout << a[i][j] << " ";
		cout << endl;
	}
	swap(a[1], a[2]);
	cout << endl;
	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 3; j++)
			cout << a[i][j] << " ";
		cout << endl;
	}
	cout << endl;

	cout << "B Matrix: " << endl;
	for (i = 0; i < 3; i++)
	{
		cout << b[i] << " ";

	}
	cout << endl << endl;


	/*
	Хід методу, де:
	a[n][n] - Матриця коефіцієнтів
	x[n], p[n] - Теперешнє и попереднє рішення
	b[n] - Стовпчик правих чисел
	*/

	for (int i = 0; i < 3; i++)
		x[i] = 1;

	cout << "Result: " << endl;
	if (diagonal(a)) {
		do
		{
			for (int i = 0; i < 3; i++)

				double var = 0;
				for (int j = 0; j < 3; j++)
					if (j != i) var += (a[i][j] * x[j]);

				x[i] = (b[i] - var) / a[i][i];
			}
			m++;
		} while (!converge(x, p, 3, eps));



		cout << "Solution:" << endl << endl;
